  ```typescript
  const [localKeyword, setLocalKeyword] = useState(filters.keyword);
  useEffect(() => {
    const timer = setTimeout(() => {
      if (localKeyword !== filters.keyword) {
        setFilters({ keyword: localKeyword });
      }
    }, 300);
    return () => clearTimeout(timer);
  }, [localKeyword]);
  ```
